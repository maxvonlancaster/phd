\section{Рівномірна відстань до простору $K_{x_{0}}(X)$ квазінеперервних у точці $x_{0}$ функцій.}


Нехай $X$ і $Y$~--- топологічні простори, $f:X\rightarrow Y$~--- відображення і $x_{0}\in X$.

\textbf{Означення}. Відображення $f:X\rightarrow Y$ називається квазінеперервним у точці $x_{0}$ з $X$, якщо для довільних околів $U$ і $V$ точок $x_{0}$ і $y_{0}=f(x_{0})$ у просторах $X$ і $Y$ відповідно існує відкрита непорожня множина $G$ в $X$, така, що $G\subseteq U$ і $f(G)\subseteq V$. $f:X\rightarrow Y$ називається квазінеперервним, якщо воно квазінеперервне в кожній точці $x$ з $X$

Зокрема, функція $f:X\rightarrow \mathbb{R}$ буде квазінеперервною у точці $x_{0}$ з $X$, якщо для довільного $\varepsilon>0$ і довільного околу $U$ точки $x_{0}$ у просторі $X$ існує непорожня множина $G$ в $X$, така, що $G\subseteq U$ і $f(G)\subseteq (f(x_{0})-\varepsilon,f(x_{0})+\varepsilon)$.

Сукупність усіх квазінеперервних у точці $x_{0}$ дійснозначних функцій $f:X\rightarrow \mathbb{R}$ ми позначаємо символом $K_{x_{0}}(X)$.
Будь-яка неперервна в точці $x_{0}\in X$ функція $f$ є квазінеперервною в цій точці. Обернене твердження не вірне. Наприклад, функція $f$, яка рівна нулю при $x=0$ та $\sin\frac{1}{x}$ при $x\neq 0$ є квазінеперервною в точці $0$, але не є неперервною в цій точці.

\begin{theorem}
\label{rvq_t1}
Нехай $b_{1} ,\, $ $b_{2} ,\, $ $b_{3} $ -- довільні дійсні числа і $f=b_{1} \chi _{(-\infty ,0)} +b_{2} \chi _{\left\{0\right\}} +b_{3} \chi _{(0,+\infty )} $. Тоді

\[d(f,\, \, K_{0} ({\mathbb R}))=d(f,\, \, S_{0} ({\mathbb R}))=\frac{1}{2} \min \left\{\left|b_{1} -b_{2} \right|,\, \left|b_{2} -b_{3} \right|\right\}.\]

Якщо $b_{1} =-1\, ,$ $b_{2} =0,\, $ $b_{3} =1,$ то $f(x)=\textmd{sgn} x$ і $d(\textmd{sgn} x, ,\, \, K_{0} ({\mathbb R}))=\frac{1}{2} $, але $d(\textmd{sgn} x, C({\mathbb R}))=1$.
\end{theorem}


\begin{proof}
Нехай $g\in S_{0}(\mathbb{R})$. Доведемо, що $$d(f,g)\geq\frac{1}{2}\min \{|b_{1}-b_{2}|,|b_{2}-b_{3}|\}$$ Припустимо, що $|g(x_{0})-b_{3}|\geq \frac{|b_{3}-b_{2}|}{2}$, для деякого $x_{0}>0$. Тоді $$d(f,g)\geq|f(x_{0})-g(x_{0})|=|b_{3}-g(x_{0})|\geq \frac{|b_{3}-b_{2}|}{2}\geq b$$ Якщо ж $|g(x_{0})-b_{1}|\geq \frac{|b_{1}-b_{2}|}{2}$, для деякого $x_{0}<0$, то: $$d(f,g)\geq |f(x_{0})-g(x_{0})|=|b_{1}-g(x_{0})|\geq \frac{|b_{1}-b_{2}|}{2}\geq b$$ Нехай тепер $|g(x)-b_{1}|<\frac{|b_{1}-b_{2}|}{2}$ при $x<0$ і $|g(x)-b_{3}|<\frac{|b_{2}-b_{3}|}{2}$ при $x>0$. Доведемо, що тоді $|g(0)-b_{2}|\geq b$. Нехай це не так, тобто $|g(0)-b_{2}|< b$. Розглянемо окіл $V=(b_{2}-b,b_{2}+b)$ точки $g(0)$. З ледь неперервності $g$ в точці $0$ випливає, що існує інтервал $G=(\alpha,\beta)$, $(\alpha<\beta)$, такий, що $g(G)\subseteq V$. А отже, існує $x^{*}\in G\setminus \{0\}$. Якщо $x^{*}>0$, то $|g(x^{*})-b_{3}|<\frac{|b_{2}-b_{3}|}{2}$ і $|g(x^{*})-b_{2}|<b.$ Тоді : $|b_{2}-b_{3}|\leq |b_{2}-g(x^{*})|+|g(x^{*})-b_{3}|<b+\frac{|b_{2}-b_{3}|}{2}\leq \frac{|b_{2}-b_{3}|}{2}+\frac{|b_{2}-b_{3}|}{2}= |b_{2}-b_{3}|$

Отримана суперечність показує, що $|g(0)-b_{2}|\geq b$. В такому разі : $$d(f,g)=\sup|f(x)-g(x)|\geq |f(0)-g(0)|=|b_{2}-g(0)|\geq b.$$ А отже : $$d(f,K_{0} ({\mathbb R}))\geq b$$ Розглянемо функцію : $$g_{0}(x)=\left\{ \begin{array}{c} \frac{b_{1}+b_{2}}{2}, \textmd{ при }x<0;\\ \frac{b_{2}+b_{3}}{2},\textmd{ при }x\geq 0  \end{array} \right. , $$  яка є ледь неперервною за побудовою, і для якої $d(f,g_{0})=b$.
\end{proof}

Аналогічно доводиться наступна теорема.

\begin{theorem}
\label{rvq_t2}
Нехай $\left|b\right|>1$, $f(x)=\sin \, \frac{1}{x} $ при $x\ne 0$ і $f(0)=b$. Тоді

\[d(f,\, \, K_{0} ({\mathbb R}))=d(f,\, \, S_{0} ({\mathbb R}))=\frac{\left|b\right|-1}{2} .\]
\end{theorem}








Нехай $E\subseteq \mathbb{R}$, $x_{0}\in E$, $f:E\rightarrow \mathbb{R}$, $D_{f}=E$ та $x_{0}$ гранична справа для $E$ (тобто $\forall \delta>0|E\cap (x_{0},x_{0}+\delta)\neq\emptyset$). Позначимо $\underline{l}=\underline{f}(x_{0}+0)=\sup\limits_{\delta>0}^{}\inf\limits_{x_{0}<x<x_{0}+\delta}^{}f(x)$, $\overline{l}=\overline{f}(x_{0}+0)=\inf\limits_{\delta>0}^{}\sup\limits_{x_{0}<x<x_{0}+\delta}^{}f(x)$, $M_{f}^{+}(x_{0};\delta)=\sup\limits_{x_{0}<x<x_{0}+\delta}^{}f(x)$ та $m_{f}^{+}(x_{0};\delta)=\inf\limits_{x_{0}<x<x_{0}+\delta}^{}f(x)$.

\begin{lemma}
\label{rvq_l1}
Нехай $f:[x_{0},+\infty)\rightarrow \mathbb{R}$,. Тоді виконується наступне:

1) $\forall\varepsilon>0$ $\exists\delta>0$ $\forall x\in(x_{0},x_{0}+\delta)\mid f(x)<\overline{f}(x_{0}+0)+\varepsilon$

2) $\forall\varepsilon>0$ $\forall\delta>0$ $\exists x\in(x_{0},x_{0}+\delta)\mid f(x)>\overline{f}(x_{0}+0)-\varepsilon $

3) $\forall\varepsilon>0$ $\exists\delta>0$ $\forall x\in(x_{0},x_{0}+\delta)\mid f(x)>\underline{f}(x_{0}+0)-\varepsilon $

4) $\forall\varepsilon>0$ $\forall\delta>0$ $\exists x\in(x_{0},x_{0}+\delta)\mid f(x)<\underline{f}(x_{0}+0)+\varepsilon$
\end{lemma}

\begin{proof}
Надалі позначимо $M_{f}^{+}(x_{0};\delta)=M_{f}$, $m_{f}^{+}(x_{0};\delta)=m_{f}$.

1) Візьмемо $\varepsilon>0$. Оскільки верхня границя $\overline{l}$ є точною нижньою межею чисел $M_{f}$, то знайдеться таке $\delta>0$, що $M_{f}<\overline{l}+\varepsilon$. Але, за означенням супремуму, $f(x)\leq M_{f}$ на $(x_{0},x_{0}+\delta)$, отже, $f(x)<\overline{l}+\varepsilon$ для всіх $x$ з $(x_{0},x_{0}+\delta)$.

2) Візьмемо $\varepsilon>0$. Оскільки верхня границя $\overline{l}$ є точною нижньою межею чисел $M_{f}$, то для довільного $\delta>0$  $M_{f}>\overline{l}-\varepsilon$. Але, так як $M_{f}$ є точною верхньою межею чисел $f(x)$ при $x\in(x_{0},x_{0}+\delta)$, то існує $x\in(x_{0},x_{0}+\delta)$, такий, що $f(x)>\overline{l}-\varepsilon$.

3) Візьмемо $\varepsilon>0$. Оскільки нижня границя $\underline{l}$ є точною верхньою межею чисел $m_{f}$, то знайдеться таке $\delta>0$, що $m_{f}>\underline{l}-\varepsilon$. Але, за означенням інфімуму, $f(x)\geq m_{f}$ на $(x_{0},x_{0}+\delta)$, а отже, $f(x)>\underline{l}-\varepsilon$ для всіх $x\in(x_{0},x_{0}+\delta)$.

4) Візьмемо $\varepsilon>0$. Оскільки нижня границя $\underline{l}$ є точною верхньою межею чисел $m_{f}$, то $m_{f}<\underline{l}+\varepsilon$ для довільного $\delta>0$. Оскільки $m_{f}$ є точною нижньою межею чисел $f(x)$, то існує $ x\in(x_{0},x_{0}+\delta)$, такий, що $f(x)<\underline{l}+\varepsilon$.
\end{proof}

\begin{lemma}
\label{rvq_l2}
1) Нехай $f:[x_{0},+\infty)\rightarrow \mathbb{R}$~--- функція, яка неперервна на $(x_{0},+\infty)$ і $\underline{l}\leq f(x_{0})\leq \overline{l}$. Тоді $f$~--- квазінеперервна в $x_{0}$.

2) Нехай $f:[x_{0},+\infty)\rightarrow \mathbb{R}$~--- функція, $f(x_{0})\notin [\underline{l},\overline{l}]$. Тоді $f$ не є квазінеперервною в точці $x_{0}$.
\end{lemma}

\begin{proof}
1) Позначимо $y_{0}=f(x_{0})$. Візьмемо довільні додатні числа $\varepsilon$ і $\delta$, і покладемо $V=(y_{0}-\varepsilon,y_{0}+\varepsilon)$ і $U=[x_{0},x_{0}+\delta)$. Покажемо, що існує таке $\xi$, що $x_{0}<\xi<x_{0}+\delta$ і $f(\xi)\in V$.

За лемою \ref{rvq_l1} існують такі точки $x^{*}$ і $x_{*}$, що
$x^{*},x_{*}\subseteq (x_{0},x_{0}+\delta)=G$, $f(x^{*})>\overline{l}-\varepsilon$, а $f(x_{*})<\underline{l}+\varepsilon$.

Припустимо, що $f(x^{*})<y_{0}+\varepsilon$. Оскільки $f(x^{*})>\overline{l}-\varepsilon$ і $\overline{l}\geq y_{0}$, то $f(x^{*})>y_{0}-\varepsilon$, отже $$y_{0}-\varepsilon<f(x^{*})<y_{0}+\varepsilon$$ належність $f(\xi)\in V$ виконується при $\xi =x^{*}$.

Нехай $f(x_{*})>y_{0}-\varepsilon$. Оскільки $f(x_{*})<\underline{l}+\varepsilon$ і $\underline{l}\leq y_{0}$, то $f(x_{*})<y_{0}+\varepsilon$, отже: $$y_{0}-\varepsilon<f(x_{*})<y_{0}+\varepsilon.$$

В такому разі $f(\xi)\in V$ виконується при $\xi =x_{*}$.

Нехай тепер $f(x^{*})\geq y_{0}+\varepsilon$ і $f(x_{*})\leq y_{0}-\varepsilon$. Розглянемо відрізок $I$ з кінцями $x_{*}$ і $x^{*}$. Оскільки $x_{0}<x^{*}<x_{0}+\delta$ та $x_{0}<x_{*}<x_{0}+\delta$, то $I\subseteq (x_{0},x_{0}+\delta)$. За умовою $D(f)={x_{0}}$, тому $I\subseteq C(f)$, тобто функція $f$ неперервна на $I$. Але $$f(x_{*})\leq y_{0}-\varepsilon<y_{0}=f(x_{0})<y_{0}+\varepsilon\leq f(x^{*}), $$ Отже, $f(x_{*})<y_{0}<f(x^{*})$. За теоремою про проміжне значення існує точка $\xi\in I$, така, що $f(\xi)=y_{0}$. Зрозуміло, що $x_{0}<\xi<x_{0}+\delta$. Таким чином, оголошена нами властивість доведена , і ми маємо таку точку $\xi$, що $x_{0}<\xi<x_{0}+\delta$ і $f(\xi)\in V$.

Оскільки і $V=(y_{0}-\varepsilon,y_{0}+\varepsilon)$ і $G=(x_{0},x_{0}+\delta)$ - відкриті інтервали, та  $V$- окіл точки $f(\xi)$, і з неперервності функції $f$ у точці $\xi$ випливає, що існує таке $\delta>0$, що $U_{0}=(\xi-\delta,\xi+\delta)\subseteq G$ і $f(U_{0})\subseteq V$. Ми знайшли відкриту непорожню множину $U_{0}$ таку, що $U_{0}\subseteq U$ і $f(U_{0})\subseteq V$, що і дає нам квазінеперервність функції $f$ у точці $x_{0}$.

2) Нехай $y_{0}\notin [\underline{l},\overline{l}]$. Доведемо, що тоді $f$ не буде квазінеперервною в точці $x_{0}$.

Припустимо, що $y_{0}>\overline{l}$. Візьмемо $\varepsilon=y_{0}-\overline{l}$ та $v=(\overline{l}+\frac{\varepsilon}{2},+\infty)$ - окіл точки $y_{0}$. Згідно з 1) Леми \ref{rvq_l1}, знайдеться $\delta>0$, таке, що для довільного $x\in (x_{0},x_{0}+\delta)|f(x)<\overline{l}+\frac{\varepsilon}{2}$. Покладемо $U=[x_{0},x_{0}+\delta)$, $G=(x_{0},x_{0}+\delta)$. Очевидно, що для довільної відкритої і непорожньої множини $U_{0}\subseteq U$ виконується $U_{0}\cap G\neq \emptyset$. Для довільної $x\in U_{0}\cap G$ $f(x)<\overline{l}+\varepsilon$, отже, $f(x)\notin V$, а тому $f(U_{0})\nsubseteq V$.

Таким чином, ми знайшли $V$ - окіл точки $y_{0}$ та $U$ - окіл $x_{0}$, такі, що для довільної відкритої та непорожньої $U_{0}\subseteq U$ виконується $f(U_{0})\nsubseteq V$ - функція $f$ не є квазінеперервною в точці $x_{0}$.

Припустимо тепер, що $y_{0}<\overline{l}$. Візьмемо $\varepsilon=\overline{l}-y_{0}$ та $V=(-\infty,\overline{l}-\frac{\varepsilon}{2})$ - окіл точки $y_{0}$. Згідно з 3) Леми \ref{rvq_l1}, знайдеться $\delta>0$, таке, що для довільного $x\in(x_{0},x_{0}+\delta)|f(x)>\underline{l}-\frac{\varepsilon}{2}$. Покладемо $U=[x_{0},x_{0}+\delta)$, $G=(x_{0},x_{0}+\delta)$. Для довільної $x\in U_{0}\cap G$ $f(x)>\underline{l}-\frac{\varepsilon}{2}$, а отже, $f(x)\notin V$, а тому $f(U_{0})\nsubseteq V$.

Таким чином, ми знайшли $V$~--- окіл точки $y_{0}$ та $U$~--- окіл точки $x_{0}$, такі, що для довільної відкритої та непорожньої $U_{0}\subseteq U$ виконується $f(U_{0})\nsubseteq V$.

Отже, ми отримали, що функція $f$ не є квазінеперервної в точці $x_{0}$.
\end{proof}





\textbf{Означення} Позначимо символом $K_{x_{0}}^{+}$ множину всіх квазінеперервних у точці $x_{0}$ функцій $f:[x_{0},+\infty)\rightarrow \mathbb{R}$, та $K_{x_{0}}^{-}$  множину всіх квазінеперервних у точці $x_{0}$ функцій $f:(-\infty,x_{0}]\rightarrow \mathbb{R}$.

\begin{theorem}
\label{rvq_t3} Нехай $f:[x_{0},+\infty)\rightarrow \mathbb{R}$~--- функція, $D(f)\subseteq{x_{0}}$ та $y_{0}=f(x_{0})\notin [\underline{l},\overline{l}]$. Тоді $$d(f,K_{x_{0}}^{+})=\frac{1}{2}\min\{\mid \underline{l}-y_{0}\mid,\mid \overline{l}-y_{0}\mid\}=\Delta$$.
\end{theorem}

\begin{proof}
Розглянемо функцію $g_{0}:[x_{0},+\infty)\rightarrow \mathbb{R}$, яка при $y_{0}>\overline{l}$ визначається як $g_{0}(x)=f(x)+\Delta$ при $x>x_{0}$ та $g_{0}(x)=y_{0}-\Delta$ при $x=x_{0}$, а при $y_{0}<\underline{l}$ визначається як $g_{0}(x)=f(x)-\Delta$ при $x>x_{0}$ та $g_{0}(x)=y_{0}+\Delta$ при $x=x_{0}$

Оскільки $\mid f(x)-g_{0}(x)\mid=\mid\pm\Delta\mid=\Delta$ для всіх $x\geq x_{0}$, то $d(f,g_{0})=\Delta$.

Доведемо, що $g_{0}$ - квазінеперервна в точці $x_{0}$.

Оскільки $f$~--- неперервна при $x>x_{0}$, то такою ж буде і функція $g_{0}$, адже $g_{0}(x)=f(x)+\Delta$ або $g_{0}(x)=f(x)-\Delta$ при $x>x_{0}$.

Якщо $y_{0}>\overline{l}$ то $0<y_{0}-\overline{l}\leq y_{0}-\underline{l}$, тому $\Delta=\frac{1}{2}(y_{0}-\overline{l})$. Знайдемо $\overline{g_{0}}(x_{0}+0)$ і $g_{0}(x_{0})$: $$\overline{g_{0}}(x_{0}+0)=\overline{\lim\limits_{x\rightarrow x_{0}+0}^{}}g_{0}(x)=\overline{\lim\limits_{x\rightarrow x_{0}+0}^{}}(f(x)+\Delta)=\overline{l}+\Delta=\overline{l}+\frac{1}{2}(y_{0}-\overline{l})=\frac{1}{2}(y_{0}+\overline{l}),$$
$$g_{0}(x_{0})=y_{0}-\Delta=y_{0}-\frac{1}{2}(y_{0}-\overline{l})=\frac{1}{2}(y_{0}+\overline{l}).$$ Таким чином, $g_{0}(x_{0})=\overline{g_{0}}(x_{0}+0)$

Якщо $y_{0}<\underline{l}$ то $\overline{l}-y_{0}\geq \underline{l}-y_{0}>0$, а отже, $\Delta=\frac{1}{2}(\underline{l}-y_{0})$. У цьому випадку: $$\underline{g_{0}}(x_{0}+0)=\underline{\lim\limits_{x\rightarrow x_{0}+0}^{}}g_{0}(x)=\underline{\lim\limits_{x\rightarrow x_{0}+0}^{}}(f(x)-\Delta)=\underline{l}-\Delta=\underline{l}-\frac{1}{2}(\underline{l}-y_{0})=\frac{1}{2}(y_{0}+\underline{l}),$$
$$g_{0}(x_{0})=y_{0}+\Delta=\frac{1}{2}(y_{0}+\underline{l}).$$ Тут також маємо $g_{0}(x_{0})=\underline{g_{0}}(x_{0}+0)$. Отже, $g_{0}(x_{0})\in [\underline{g_{0}}(x_{0}+0),\overline{g_{0}}(x_{0}+0)]$, звідси, за лемою \ref{rvq_l2} маємо, що $g_{0}$ є квазінеперервною в точці $x_{0}$.

Таким чином, $g_{0}\in K_{x_{0}}^{+}$ і $d(f,g_{0})=\Delta$. Доведемо тепер, що ця відстань є мінімальною, тобто, що для довільного $g\in K_{x_{0}}^{+}$, $d(f,g_{0})\geq\Delta$.

Припустимо, що $g(x)>f(x)+\Delta$ для деякого $x>x_{0}$. Тоді $$d(f,g)=|| f-g||\geq \mid g(x)-f(x)\mid=g(x)-f(x)>\Delta,$$ бо $g(x)-f(x)>\Delta>0.$

Припустимо тепер, що $g(x)<f(x)-\Delta$ для деякого $x>x_{0}$. Тоді $$d(f,g)=|| f-g||\geq \mid g(x)-f(x)\mid=f(x)-g(x)>\Delta,$$ бо $f(x)-g(x)>\Delta>0.$

Нехай тепер $f(x)-\Delta\leq g(x)\leq f(x)+\Delta$ для кожного $x>x_{0}$. Тоді $\overline{g_{0}}(x_{0}+0)=\overline{\lim\limits_{x\rightarrow x_{0}+0}^{}}g_{0}(x)=\overline{\lim\limits_{x\rightarrow x_{0}+0}^{}}(f(x)+\Delta)=\overline{l}+\Delta$ і $\underline{g_{0}}(x_{0}+0)=\underline{\lim\limits_{x\rightarrow x_{0}+0}^{}}g_{0}(x)=\underline{\lim\limits_{x\rightarrow x_{0}+0}^{}}(f(x)-\Delta)=\underline{l}-\Delta$. За лемою \ref{rvq_l2} з квазінеперервності функції $g$ в точці $x_{0}$ випливає, що $$\underline{g_{0}}(x_{0}+0)\leq g(x_{0})\leq \overline{g_{0}}(x_{0}+0).$$ Тому $\underline{l}-\Delta\leq g(x_{0})\leq \overline{l}+\Delta$. В такому разі $$d(f,g)=|| f-g||\geq \mid g(x_{0})-f(x_{0})\mid=f(x_{0})-g(x_{0})\geq y_{0}-\overline{l}-\Delta=2\Delta-\Delta=\Delta,$$ якщо $y_{0}>\overline{l}$, і $$d(f,g)=|| f-g||\geq \mid g(x_{0})-f(x_{0})\mid=g(x_{0})-f(x_{0})\geq \underline{l}-\Delta-y_{0}=2\Delta-\Delta=\Delta,$$ якщо $y_{0}<\underline{l}$.

Ми показали, що для кожної функції $g\in K_{x_{0}}^{+}$ виконується нерівність $|| f-g||\geq \Delta$, а для функції $g_{0}\in K_{x_{0}}^{+}$ має місце рівність $|| f-g||=\Delta$. Тому $d(f,K_{x_{0}}^{+})=\Delta$.
\end{proof}

\begin{lemma}
\label{rvq_l3} Нехай $f:X\rightarrow Y$~--- неперервне сюр'єктивне і відкрите відображення на $X$, $g:Y\rightarrow Z$~--- квазінеперервне в точці $y_{0}=f(x_{0})$ відображення. Тоді $h=g\circ f$~--- квазінеперервне в точці $x_{0}$.
\end{lemma}

\begin{proof} Нехай $W$~--- окіл точки $z_{0}=h(x_{0})$, та $U$~--- відкритий окіл точки $x_{0}$. $V=f(U)$~--- відкритий, оскільки $f$~--- відкрите. Оскільки відображення $g$~--- квазінеперервне, то існує $ H$ - відкрита і непорожня множина, така, що $H\subseteq V$ і $g(H)\subseteq W$.

Оскільки $H$~--- відкрита і $f$~--- неперервне, то $f^{-1}(H)$~--- відкрите. Позначимо $G=f^{-1}(H)\cap U$. $G$~--- відкрита як перетин двох відкритих множин, та непорожня, оскільки $f$~--- сюр'єктивна. Окрім того, $h(G)=g(f(f^{-1}(H)\cap U))=g(H)\subseteq W$. Отже, для довільного $W$~--- околу точки $z_{0}=h(x_{0})$ і довільного $U$~--- околу точки $x_{0}$ існує $G$~--- відкрита і непорожня, така, що $G\subseteq U$ і $h(G)\subseteq W$. Ми отримали, що $h$ є квазінеперервною в точці $x_{0}$.
\end{proof}

\begin{theorem}
\label{rvq_t4} Нехай $f:(-\infty,x_{0}]\rightarrow\mathbb{R}$~--- функція, $D(f)\subseteq{x_{0}}$ та $y_{0}=f(x_{0})\notin[\underline{l}=,\overline{l}$. Тоді $$d(f,K_{x_{0}}^{-}=\frac{1}{2}\min\{\mid \underline{l}-y_{0}\mid,\mid \overline{l}-y_{0}\mid\}=\Delta$$.
\end{theorem}

\begin{proof} Позначимо $h=f\circ \varphi$, де $\varphi(x)=2x_{0}-x$~--- неперервна, сюр'єктивна і відкрита функція на $\mathbb{R}$. Знайдемо $h(x_{0})$, $\overline{h}(x_{0}+0)$ та $\underline{h}(x_{0}+0)$: $$h(x_{0})=f(\varphi(x_{0}))=f(x_{0})=y_{0}$$ $$\overline{h}(x_{0}+0)=\inf\limits_{\delta>0}^{}\sup\limits_{x_{0}<x<x_{0}+\delta}^{}h(x)=$$ $$=\inf\limits_{\delta>0}^{}\sup\limits_{x_{0}<x<x_{0}+\delta}^{}f(\varphi(x))=\inf\limits_{\delta>0}^{}\sup\limits_{x_{0}-\delta<\varphi(x)<x_{0}}^{}f(\varphi(x))=\overline{f}(x_{0}-0);$$
оскільки при $x_{0}<x<x_{0}+\delta$, $x_{0}-\delta<\varphi(x)<x_{0}$.

Аналогічно : $$\underline{h}(x_{0}+0)=\sup\limits_{\delta>0}^{}\inf\limits_{x_{0}<x<x_{0}+\delta}^{}h(x)=$$ $$=\sup\limits_{\delta>0}^{}\inf\limits_{x_{0}<x<x_{0}+\delta}^{}f(\varphi(x))=\sup\limits_{\delta>0}^{}\inf\limits_{x_{0}-\delta<\varphi(x)<x_{0}}^{}f(\varphi(x))=\overline{f}(x_{0}-0).$$ Згідно з теоремою \ref{rvq_t3} маємо, що : $$d(h,K_{x_{0}}^{+})=\frac{1}{2}\min{|h(x_{0})-\overline{h}(x_{0}+0)|,|h(x_{0})-\underline{h}(x_{0}+0)|}=$$$$=\frac{1}{2}\min{\mid \underline{l}-y_{0}\mid,\mid \overline{l}-y_{0}\mid}=\Delta.$$ Виберемо довільну $g\in K_{x_{0}}^{+}$, та порахуємо $d(h,g)$: $$d(h,g)=\sup\limits_{x\in[x_{0},+\infty)}^{}|h(x)-g(x)|=\sup\limits_{x^{*}\in(-\infty,x_{0}]}^{}|h(\varphi(x))-g(\varphi(x))|=$$$$=\sup\limits_{x^{*}\in(-\infty,x_{0}]}^{}|f(x^{*})-g^{*}(x^{*})|=d(f,g^{*}),$$ де $g^{*}=g\circ\varphi:(-\infty,x_{0}]\rightarrow\mathbb{R}$ - квазінеперервна за лемою \ref{rvq_l3}, $g^{*}\in K_{x_{0}}^{-}$.

Отже, для довільної $g\in K_{x_{0}}^{+}$ $d(h,g)=d(f,g^{*})$, де $g^{*}=g\circ\varphi \in K_{x_{0}}^{-}$.

З даної ріності маємо, що $$d(f,K_{x_{0}}^{-})=d(h,K_{x_{0}}^{+})=\Delta.$$
\end{proof}

\textbf{Означення} Позначимо через $K^{*}_{x_{0}}(\mathbb{R})$ сукупність усіх двосторонньо квазінеперервних в точці $x_{0}$ функцій $f:\mathbb{R}\rightarrow\mathbb{R}$.
